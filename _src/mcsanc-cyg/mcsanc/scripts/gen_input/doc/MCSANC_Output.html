def initialize(path,tag)
  file="#{path}/mcsanc-#{tag}-output.txt"
  @results=OpenStruct.new
  @config=OpenStruct.new
  @results.born=OpenStruct.new
  lines=File.open(file,"r").readlines
  lines.each{ |line| 
    if line.match /\s*born\s*\|.*\|/ then 
      @results.born.value, @results.born.error=line.scanf("%s %s %d %f %f").pop(2).map{|d| d.to_f}
    end and break
  }
end
